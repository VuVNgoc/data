# Mapper function
def mapper(line):
    parts = line.split()
    year = parts[0]
    values = list(map(int, parts[1:]))
    avg_value = sum(values) / len(values)
    return year, avg_value

# Reducer function
def reducer(mapped_data):
    for year, avg_value in mapped_data:
        if avg_value > 30:
            print(f"{year}: {avg_value:.2f}")

# Main function to process data
def main(file_path):
    with open(file_path, 'r') as f:
        data = f.readlines()

    mapped_data = [mapper(line) for line in data]
    reducer(mapped_data)

# Run the program
main('sample.txt')
